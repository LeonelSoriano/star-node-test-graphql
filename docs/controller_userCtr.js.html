"use strict";

import models from "../models";

import logger from "../until/logger";

/**
 * Modelo que posee la entidad de usuario.
 */
class UserCtr {

    /**
    * @return {Object} regresa todas los usuarios
    */
    async listUser(){
        try{
            let users = await models.User.find({});
            return users;
        }catch(err){
            logger.log("error", err.stack);
            return [];
        }
    }
}


export {UserCtr};
